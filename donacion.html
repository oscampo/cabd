- paths: - https://oscampo.github.io/cabd/data/people.csv - https://oscampo.github.io/cabd/data/people.sqlite import sqlite3 conn = sqlite3.connect('people.sqlite') cursor = conn.execute("SELECT * from people") for row in cursor.fetchall(): print("Nombre", row[0], "Picture", row[1]) print("Altura", row[2], "Peso", row[3]) print("Género", row[4], "Fecha de Nacimiento", row[5]) print("Biografía", row[6], "\n") conn.close()
